If that URL is unreachable, create a minimal placeholder PNG using Python's `struct`
module to write a 1x1 white pixel PNG (valid PNG format, minimal bytes):

```python
import zlib, struct

def write_minimal_png(path):
    def chunk(name, data):
        c = zlib.crc32(name + data) & 0xffffffff
        return struct.pack('>I', len(data)) + name + data + struct.pack('>I', c)
    png = b'\x89PNG\r\n\x1a\n'
    png += chunk(b'IHDR', struct.pack('>IIBBBBB', 1, 1, 8, 2, 0, 0, 0))
    raw = b'\x00\xff\xff\xff'
    png += chunk(b'IDAT', zlib.compress(raw))
    png += chunk(b'IEND', b'')
    with open(path, 'wb') as f:
        f.write(png)

write_minimal_png('/home/pigotz/Projects/drum-theory/public/drum-sheet.png')
```
